# 常用命令




~~~
#启动服务
redis-server [配置文件路径]

# 连接客户端
./redis-cli -h hostname -p portName

# 查看最大客户端连接数
config get maxclients

# 查看内存信息
info memory

# 查看客户端信息
info clients

key *
